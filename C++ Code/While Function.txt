#include<iostream>

using namespace std;

int main()
{
int number=0,mitun;
cout<<"Finding odd and even numbers~";

cin>>number;
for(int i=1;i<=number;i++)
{
	if(i%2==0)
	{
		cout<<i<<":is even\n";
	}
	else
	{
	cout<<i<<"is odd\n";
	}
}
cin>>mitun;
return 0;

}
